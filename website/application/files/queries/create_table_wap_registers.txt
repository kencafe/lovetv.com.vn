CREATE TABLE IF NOT EXISTS `[[is_tables_name]]` (
  `id` bigint(20) unsigned NOT NULL AUTO_INCREMENT,
  `registerId` varchar(36) DEFAULT NULL COMMENT 'ID của Register, YmdHis + random 10 số',
  `serviceId` varchar(36) DEFAULT NULL COMMENT 'Service phát sinh đăng ký',
  `packageId` varchar(36) DEFAULT NULL COMMENT 'Gói cước đăng ký',
  `commandId` varchar(36) DEFAULT NULL COMMENT 'Mã đăng ký',
  `msisdn` varchar(20) DEFAULT NULL COMMENT 'SĐT đăng ký',
  `otpId` varchar(36) DEFAULT NULL COMMENT 'OTP ID',
  `otpCode` varchar(6) DEFAULT NULL COMMENT 'OTP Code',
  `channel` varchar(36) DEFAULT NULL COMMENT 'Kênh phát sinh',
  `application` varchar(36) DEFAULT NULL COMMENT 'Dùng phân biệt đăng ký thật hay PTTB',
  `note` varchar(128) DEFAULT NULL COMMENT 'ghi chú thêm',
  `reason` varchar(255) DEFAULT NULL COMMENT 'Lý do phát sinh log',
  `data` varchar(1024) DEFAULT NULL COMMENT 'Data kèm theo Request',
  `status` tinyint(1) DEFAULT '0' COMMENT 'Trạng thái: 0: Chờ kích hoạt, 1: Kích hoạt, 2: Lỗi',
  `created_at` datetime DEFAULT NULL COMMENT 'Thời gian tạo',
  `updated_at` datetime DEFAULT NULL COMMENT 'Thời gian cập nhật',
  `logs` text,
  PRIMARY KEY (`id`),
  UNIQUE KEY `registerId` (`registerId`),
  UNIQUE KEY `otpId` (`otpId`),
  KEY `serviceId` (`serviceId`),
  KEY `packageId` (`packageId`),
  KEY `commandId` (`commandId`),
  KEY `msisdn` (`msisdn`),
  KEY `status` (`status`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='Bảng lưu trữ Wap Registers';
