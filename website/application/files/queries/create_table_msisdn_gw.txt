CREATE TABLE IF NOT EXISTS `[[is_tables_name]]` (
  `id` bigint(20) unsigned NOT NULL AUTO_INCREMENT,
  `cpId` varchar(36) DEFAULT NULL COMMENT 'Là ID của CP cung cấp bởi VNM',
  `returnUrl` varchar(255) DEFAULT NULL COMMENT 'Url encoded của CP mà request của khách hàng sẽ được redirect đến sau khi VNM xử lý',
  `redirectUrl` varchar(255) DEFAULT NULL COMMENT 'URL redirect đến sau khi xử lý nhận diện',
  `requestId` varchar(36) DEFAULT NULL COMMENT 'ID duy nhất mà CP gán cho mỗi request của khách hàng, sinh bằng YmdHis + random 10 số',
  `requestTime` varchar(72) DEFAULT NULL COMMENT 'Thời gian khách hàng request đến website của CP theo chuẩn ISO, ví dụ: 2015-0706T22:54:50Z',
  `requestSignature` varchar(128) DEFAULT NULL COMMENT 'Chuỗi bảo mật khi Request tránh giả mạo thông tin',
  `responseCode` varchar(128) DEFAULT NULL COMMENT 'Trạng thái giao dịch do VNM trả về. Giá trị là “0” (zero) cho biết đã xử lý thành công. Tất cả các giá trị khác là thất bại',
  `responseTime` varchar(72) DEFAULT NULL COMMENT 'Thời gian VNM GW trả về theo chuẩn ISO, ví dụ: 2015-07-06T22:54:50Z',
  `responseTransactionId` varchar(128) DEFAULT NULL COMMENT 'Mã giao dịch duy nhất được sinh ra từ VNM, dùng để tra soát giao dịch với CP',
  `responseMessage` varchar(2048) DEFAULT NULL COMMENT 'Chứa tham số MSISDN nếu truy vấn thành công, chứa các thông tin khác nếu thất bại',
  `responseAdditionData` varchar(1024) DEFAULT NULL COMMENT 'Thông tin thêm nếu có',
  `responseSignature` varchar(128) DEFAULT NULL COMMENT 'Trường này cho phép CP kiểm tra bản tin trả về từ VNM có hợp lệ hay không.',
  `status` tinyint(1) DEFAULT '0' COMMENT 'Trạng thái: 0: Request, 1: Response Success, 2: Response Error',
  `created_at` datetime DEFAULT NULL COMMENT 'Thời gian tạo',
  `updated_at` datetime DEFAULT NULL COMMENT 'Thời gian cập nhật',
  `logs` text,
  PRIMARY KEY (`id`),
  UNIQUE KEY `requestId` (`requestId`),
  UNIQUE KEY `responseTransactionId` (`responseTransactionId`),
  KEY `cpId` (`cpId`),
  KEY `responseCode` (`responseCode`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
